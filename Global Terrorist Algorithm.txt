BEGIN
INT COMPARE(A, B)
    LENGTH = LEN(A)
    FOR (I = 0; I < LENGTH; I += 1)
        IF (A[I] != B[I])
            RETURN 0
   RETURN 1
END

BEGIN
VOID GLOBAL_TERRORIST()
CREATE TOTAL_TERROR
OPEN FILE TERROR_TEXT from Terrorist.txt
TOTAL_TERROR = (INT) READLINE 0 from TERROR_TEXT
CREATE a MULTIDIMENTIONAL ARRAY of SIZE TOTAL_TERROR for the dataset called TERRORIST
CREATE an ARRAY of SIZE TOTAL_TERROR for the sum of features in the dataset called TERROR_SUM
CREATE an ARRAY of SIZE TOTAL_TERROR for SIMILARITY_ID
FOR (COUNTER = 1; COUNTER <= TOTAL_TERROR; COUNTER += 1) 
    TERRORIST[COUNTER - 1] = READLINE COUNTER from TERROR_TEXT and SPLIT BY ','
    TERRORSUM[COUNTER - 1] = SUM OF TERRORIST[COUNTER - 1]
ENDFOR
FOR (I = 0; I < TOTAL_TERROR; I += 1)
    FOR (J = 0; J < TOTAL_TERROR; J += 1)
        IF (I == J)
            SKIP
        ENDIF
        IF (TERROR_SUM[I] == TERROR_SUM[J])
            CREATE SAME_TERROR = COMPARE(TERRORIST[I], TERRORIST[J])
            IF (SAME_TERROR == 1)
                SIMILARITY_ID[I] = J + 1
            ENDIF
        ELSE 
            SIMILARITY_ID[I] = NULL
        ENDIF
    ENDFOR
ENDFOR
CLOSE FILE TERROR_TEXT
DISPLAY SIMILARITY_ID
END
